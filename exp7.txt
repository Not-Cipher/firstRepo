#!/bin/bash

echo "Enter a number:"
read num

echo "1. Even/Odd"
echo "2. Prime"
echo "3. Palindrome"
echo "4. Armstrong"
read choice

case $choice in
1)
if [ $((num%2)) -eq 0 ]; then
    echo "Even Number"
else
    echo "Odd Number"
fi
;;

2)
flag=0
for ((i=2;i<=num/2;i++))
do
    if [ $((num%i)) -eq 0 ]; then
        flag=1
    fi
done

if [ $flag -eq 0 ]; then
    echo "Prime Number"
else
    echo "Not Prime"
fi
;;

3)
rev=0
temp=$num
while [ $temp -gt 0 ]
do
    rem=$((temp%10))
    rev=$((rev*10 + rem))
    temp=$((temp/10))
done

if [ $rev -eq $num ]; then
    echo "Palindrome"
else
    echo "Not Palindrome"
fi
;;

4)
sum=0
temp=$num
while [ $temp -gt 0 ]
do
    rem=$((temp%10))
    sum=$((sum + rem*rem*rem))
    temp=$((temp/10))
done

if [ $sum -eq $num ]; then
    echo "Armstrong Number"
else
    echo "Not Armstrong"
fi
;;

*)
echo "Invalid Choice"
;;
esac








#!/bin/bash

echo "Enter elements:"
read -a arr

echo "Enter element to search:"
read key

found=0

for i in "${arr[@]}"
do
    if [ "$i" == "$key" ]; then
        found=1
    fi
done

if [ $found -eq 1 ]; then
    echo "Element found"
else
    echo "Element not found"
fi
