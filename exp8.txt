#!/bin/bash

echo "Enter file name:"
read file

if [ -f "$file" ]; then
    lines=$(grep -c "" "$file")
    words=$(wc -w < "$file")
    chars=$(wc -c < "$file")

    echo "Number of lines: $lines"
    echo "Number of words: $words"
    echo "Number of characters: $chars"
else
    echo "File does not exist"
fi
